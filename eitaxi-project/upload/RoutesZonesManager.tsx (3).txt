"use client";

import { useState, useEffect, useMemo } from "react";
import {
  Route,
  Plus,
  X,
  Loader2,
  MapPin,
  Globe,
  ChevronDown,
  ChevronRight,
  Search,
  Plane,
  Train,
  Mountain,
  Building,
  Edit3,
} from "lucide-react";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import { Separator } from "@/components/ui/separator";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@/components/ui/collapsible";
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import {
  SWISS_CANTONS,
  LIECHTENSTEIN,
  POPULAR_PLACES,
  getMunicipalitiesByCanton,
  getMunicipalitiesByDistrict,
} from "@/lib/geo-data";

interface ServiceZone {
  id: string;
  zoneName: string;
  zoneType: string;
  zoneMode: string;  // "pickup" = recogida, "service" = destino
  exclusions: string[];
  isActive: boolean;
}

interface DriverRoute {
  id: string;
  origin: string;
  destination: string;
  originType: string;
  destType: string;
  isActive: boolean;
}

interface RoutesZonesManagerProps {
  driverId: string;
  baseCity: string;
  baseCanton: string;
}

export default function RoutesZonesManager({ driverId, baseCity, baseCanton }: RoutesZonesManagerProps) {
  const [zones, setZones] = useState<ServiceZone[]>([]);
  const [routes, setRoutes] = useState<DriverRoute[]>([]);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [activeTab, setActiveTab] = useState<"zones" | "routes">("zones");
  
  // Form state for zones - hierarchical selection
  const [selectedCanton, setSelectedCanton] = useState<string>("");
  const [selectedDistrict, setSelectedDistrict] = useState<string>("");
  const [selectedMunicipality, setSelectedMunicipality] = useState<string>("");
  const [customZoneName, setCustomZoneName] = useState("");
  const [newZoneMode, setNewZoneMode] = useState<string>("service"); // "pickup" o "service"
  const [newExclusions, setNewExclusions] = useState<string[]>([]);
  const [showExclusions, setShowExclusions] = useState(false);
  const [searchTerm, setSearchTerm] = useState("");
  
  // Edit zone state
  const [editingZone, setEditingZone] = useState<ServiceZone | null>(null);
  const [editExclusions, setEditExclusions] = useState<string[]>([]);
  const [showEditModal, setShowEditModal] = useState(false);
  
  // Expanded state for canton accordions
  const [expandedCantons, setExpandedCantons] = useState<Set<string>>(new Set());
  
  // Form state for routes
  const [newOrigin, setNewOrigin] = useState("");
  const [newOriginType, setNewOriginType] = useState("city");
  const [newDestination, setNewDestination] = useState("");
  const [newDestType, setNewDestType] = useState("city");

  // Fetch zones and routes
  useEffect(() => {
    const fetchData = async () => {
      try {
        const [zonesRes, routesRes] = await Promise.all([
          fetch(`/api/driver/zones?driverId=${driverId}`),
          fetch(`/api/driver/routes?driverId=${driverId}`),
        ]);
        
        const zonesData = await zonesRes.json();
        const routesData = await routesRes.json();
        
        if (zonesData.success) setZones(zonesData.zones);
        if (routesData.success) setRoutes(routesData.routes);
      } catch (error) {
        console.error("Error fetching data:", error);
      } finally {
        setLoading(false);
      }
    };
    
    fetchData();
  }, [driverId]);

  // Get available districts based on selected canton
  const availableDistricts = useMemo(() => {
    if (!selectedCanton) return [];
    if (selectedCanton === "liechtenstein") {
      return [{ id: "li-all", name: "Todo Liechtenstein", municipalities: LIECHTENSTEIN.municipalities }];
    }
    const canton = SWISS_CANTONS.find(c => c.id === selectedCanton);
    return canton?.districts || [];
  }, [selectedCanton]);

  // Get available municipalities for exclusion
  const availableMunicipalities = useMemo(() => {
    if (selectedCanton === "liechtenstein") {
      return LIECHTENSTEIN.municipalities;
    }
    if (selectedCanton && selectedDistrict) {
      return getMunicipalitiesByDistrict(selectedCanton, selectedDistrict);
    }
    if (selectedCanton) {
      return getMunicipalitiesByCanton(selectedCanton);
    }
    return [];
  }, [selectedCanton, selectedDistrict]);

  // Get zone name for display
  const getZoneName = () => {
    if (customZoneName) return customZoneName;
    if (selectedMunicipality && selectedMunicipality !== "__all__") return selectedMunicipality;
    if (selectedDistrict && selectedDistrict !== "__all__") {
      const canton = SWISS_CANTONS.find(c => c.id === selectedCanton);
      const district = canton?.districts.find(d => d.id === selectedDistrict);
      return district?.name || "";
    }
    if (selectedCanton === "liechtenstein") return "Liechtenstein";
    if (selectedCanton) {
      const canton = SWISS_CANTONS.find(c => c.id === selectedCanton);
      return canton?.name || "";
    }
    return "";
  };

  // Get zone type
  const getZoneType = (): string => {
    if (selectedMunicipality && selectedMunicipality !== "__all__") return "municipality";
    if (selectedDistrict && selectedDistrict !== "__all__") return "district";
    if (selectedCanton === "liechtenstein") return "country";
    if (selectedCanton) return "canton";
    return "custom";
  };

  // Toggle canton expansion
  const toggleCantonExpansion = (cantonId: string) => {
    const newExpanded = new Set(expandedCantons);
    if (newExpanded.has(cantonId)) {
      newExpanded.delete(cantonId);
    } else {
      newExpanded.add(cantonId);
    }
    setExpandedCantons(newExpanded);
  };

  // Add zone
  const handleAddZone = async () => {
    const zoneName = getZoneName();
    if (!zoneName) return;
    
    setSaving(true);
    try {
      const res = await fetch("/api/driver/zones", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          driverId,
          zoneName,
          zoneType: getZoneType(),
          zoneMode: newZoneMode,  // "pickup" o "service"
          exclusions: newExclusions,
        }),
      });
      
      const data = await res.json();
      if (data.success) {
        setZones([...zones, data.zone]);
        // Reset form
        setSelectedCanton("");
        setSelectedDistrict("");
        setSelectedMunicipality("");
        setCustomZoneName("");
        setNewExclusions([]);
        setShowExclusions(false);
        setSearchTerm("");
      }
    } catch (error) {
      console.error("Error adding zone:", error);
    } finally {
      setSaving(false);
    }
  };

  // Delete zone
  const handleDeleteZone = async (zoneId: string) => {
    try {
      await fetch(`/api/driver/zones?zoneId=${zoneId}`, { method: "DELETE" });
      setZones(zones.filter(z => z.id !== zoneId));
    } catch (error) {
      console.error("Error deleting zone:", error);
    }
  };

  // Update zone exclusions
  const handleUpdateZoneExclusions = async () => {
    if (!editingZone) return;
    
    setSaving(true);
    try {
      const res = await fetch("/api/driver/zones", {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          driverId,
          zoneId: editingZone.id,
          exclusions: editExclusions,
        }),
      });
      
      const data = await res.json();
      if (data.success) {
        setZones(zones.map(z => z.id === editingZone.id ? data.zone : z));
        setShowEditModal(false);
        setEditingZone(null);
        setEditExclusions([]);
      }
    } catch (error) {
      console.error("Error updating zone:", error);
    } finally {
      setSaving(false);
    }
  };

  // Add route
  const handleAddRoute = async () => {
    if (!newOrigin || !newDestination) return;
    
    setSaving(true);
    try {
      const res = await fetch("/api/driver/routes", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          driverId,
          origin: newOrigin,
          destination: newDestination,
          originType: newOriginType,
          destType: newDestType,
        }),
      });
      
      const data = await res.json();
      if (data.success) {
        setRoutes([...routes, data.route]);
        setNewOrigin("");
        setNewDestination("");
        setNewOriginType("city");
        setNewDestType("city");
      }
    } catch (error) {
      console.error("Error adding route:", error);
    } finally {
      setSaving(false);
    }
  };

  // Delete route
  const handleDeleteRoute = async (routeId: string) => {
    try {
      await fetch(`/api/driver/routes?routeId=${routeId}`, { method: "DELETE" });
      setRoutes(routes.filter(r => r.id !== routeId));
    } catch (error) {
      console.error("Error deleting route:", error);
    }
  };

  // Quick add zone
  const quickAddZone = async (zoneName: string, zoneType: string, zoneMode: string = "service", exclusions: string[] = []) => {
    setSaving(true);
    try {
      const res = await fetch("/api/driver/zones", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          driverId,
          zoneName,
          zoneType,
          zoneMode,  // "pickup" o "service"
          exclusions,
        }),
      });
      
      const data = await res.json();
      if (data.success) {
        setZones([...zones, data.zone]);
      }
    } catch (error) {
      console.error("Error adding zone:", error);
    } finally {
      setSaving(false);
    }
  };

  // Filter cantons by search
  const filteredCantons = useMemo(() => {
    if (!searchTerm) return SWISS_CANTONS;
    const term = searchTerm.toLowerCase();
    return SWISS_CANTONS.filter(c => 
      c.name.toLowerCase().includes(term) ||
      c.nameDE.toLowerCase().includes(term) ||
      c.code.toLowerCase().includes(term)
    );
  }, [searchTerm]);

  // Computed zones by mode
  const pickupZones = useMemo(() => zones.filter(z => z.zoneMode === 'pickup'), [zones]);
  const serviceZones = useMemo(() => zones.filter(z => z.zoneMode === 'service'), [zones]);

  if (loading) {
    return (
      <div className="flex items-center justify-center p-8">
        <Loader2 className="h-8 w-8 animate-spin text-yellow-400" />
      </div>
    );
  }

  return (
    <Card className="border-border bg-card">
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Globe className="h-5 w-5 text-yellow-400" />
          Zonas y Rutas
        </CardTitle>
        <p className="text-sm text-muted-foreground">
          Define dónde trabajas y qué rutas haces. Los clientes te encontrarán según estas zonas.
        </p>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Tab selector */}
        <div className="flex gap-2">
          <Button
            variant={activeTab === "zones" ? "default" : "outline"}
            onClick={() => setActiveTab("zones")}
            className={activeTab === "zones" ? "bg-yellow-400 text-black" : ""}
          >
            <MapPin className="h-4 w-4 mr-2" />
            Zonas de servicio
          </Button>
          <Button
            variant={activeTab === "routes" ? "default" : "outline"}
            onClick={() => setActiveTab("routes")}
            className={activeTab === "routes" ? "bg-yellow-400 text-black" : ""}
          >
            <Route className="h-4 w-4 mr-2" />
            Rutas específicas
          </Button>
        </div>

        {/* ZONES TAB */}
        {activeTab === "zones" && (
          <div className="space-y-4">
            {/* Zonas de RECOGIDA */}
            <div className="p-4 rounded-lg bg-green-500/10 border border-green-500/30">
              <div className="flex items-center gap-2 mb-2">
                <MapPin className="h-5 w-5 text-green-400" />
                <h3 className="font-semibold text-green-400">¿Desde dónde vas a buscar clientes?</h3>
              </div>
              <p className="text-sm text-muted-foreground mb-3">
                Los clientes de estas zonas te verán cuando busquen un taxi.
              </p>
              <p className="text-xs text-green-300 bg-green-500/10 p-2 rounded mb-3">
                💡 Ejemplo: Si añades "Schaan", aparecerás cuando alguien en Schaan busque un taxi.
              </p>
              
              {/* Pickup zones list */}
              {pickupZones.length > 0 && (
                <div className="space-y-2 mb-3">
                  <Label className="text-xs text-green-400">Tus zonas de recogida:</Label>
                  <div className="max-h-32 overflow-y-auto space-y-1">
                    {pickupZones.map((zone) => (
                      <div key={zone.id} className="flex items-center justify-between p-2 rounded bg-green-500/10 border border-green-500/20">
                        <div className="flex items-center gap-2">
                          <Badge variant="outline" className="text-xs border-green-500 text-green-400">
                            {zone.zoneType}
                          </Badge>
                          <span className="text-sm">{zone.zoneName}</span>
                        </div>
                        <Button variant="ghost" size="sm" onClick={() => handleDeleteZone(zone.id)} className="text-red-400">
                          <X className="h-3 w-3" />
                        </Button>
                      </div>
                    ))}
                  </div>
                </div>
              )}
              
              {/* Quick add pickup zones */}
              <div className="flex flex-wrap gap-2">
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => quickAddZone("Liechtenstein", "country", "pickup")}
                  className="text-xs border-green-500/50 text-green-400 hover:bg-green-500/10"
                >
                  + Liechtenstein
                </Button>
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => quickAddZone(baseCity, "municipality", "pickup")}
                  className="text-xs border-green-500/50 text-green-400 hover:bg-green-500/10"
                >
                  + {baseCity}
                </Button>
              </div>
            </div>

            {/* Zonas de DESTINO */}
            <div className="p-4 rounded-lg bg-yellow-500/10 border border-yellow-500/30">
              <div className="flex items-center gap-2 mb-2">
                <Globe className="h-5 w-5 text-yellow-400" />
                <h3 className="font-semibold text-yellow-400">🎯 ¿A dónde llevas a tus clientes?</h3>
              </div>
              <p className="text-sm text-muted-foreground mb-3">
                Aparecerás para viajes que terminen en estas zonas.
              </p>
              <p className="text-xs text-yellow-300 bg-yellow-500/10 p-2 rounded mb-3">
                💡 Ejemplo: Si añades "Aeropuerto Zúrich", aparecerás cuando alguien quiera ir al aeropuerto desde {baseCity}.
              </p>
              
              {/* Service zones list */}
              {serviceZones.length > 0 && (
                <div className="space-y-2 mb-3">
                  <Label className="text-xs text-yellow-400">Tus zonas de destino:</Label>
                  <div className="max-h-32 overflow-y-auto space-y-1">
                    {serviceZones.map((zone) => (
                      <div key={zone.id} className="flex items-center justify-between p-2 rounded bg-yellow-500/10 border border-yellow-500/20">
                        <div className="flex items-center gap-2">
                          <Badge variant="outline" className="text-xs border-yellow-500 text-yellow-400">
                            {zone.zoneType}
                          </Badge>
                          <span className="text-sm">{zone.zoneName}</span>
                          {zone.exclusions.length > 0 && (
                            <span className="text-xs text-red-400">❌ {zone.exclusions.length}</span>
                          )}
                        </div>
                        <div className="flex gap-1">
                          <Button variant="ghost" size="sm" onClick={() => { setEditingZone(zone); setEditExclusions([...zone.exclusions]); setShowEditModal(true); }}>
                            <Edit3 className="h-3 w-3" />
                          </Button>
                          <Button variant="ghost" size="sm" onClick={() => handleDeleteZone(zone.id)} className="text-red-400">
                            <X className="h-3 w-3" />
                          </Button>
                        </div>
                      </div>
                    ))}
                  </div>
                </div>
              )}
              
              {/* Quick add service zones */}
              <div className="flex flex-wrap gap-2">
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => quickAddZone("Liechtenstein", "country", "service")}
                  className="text-xs border-yellow-500/50 text-yellow-400 hover:bg-yellow-500/10"
                >
                  + Liechtenstein
                </Button>
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => quickAddZone("Aeropuerto Zúrich", "place", "service")}
                  className="text-xs border-yellow-500/50 text-yellow-400 hover:bg-yellow-500/10"
                >
                  + Aeropuerto Zúrich
                </Button>
                <Button
                  variant="outline"
                  size="sm"
                  onClick={() => quickAddZone("Aeropuerto Ginebra", "place", "service")}
                  className="text-xs border-yellow-500/50 text-yellow-400 hover:bg-yellow-500/10"
                >
                  + Aeropuerto Ginebra
                </Button>
              </div>
            </div>

            <Separator />

            {/* Add new zone form */}
            <div className="space-y-3">
              <Label className="text-sm font-semibold">Añadir nueva zona personalizada:</Label>
              
              {/* Zone mode selector */}
              <div className="flex gap-2">
                <Button
                  variant={newZoneMode === "pickup" ? "default" : "outline"}
                  size="sm"
                  onClick={() => setNewZoneMode("pickup")}
                  className={newZoneMode === "pickup" ? "bg-green-500 text-white" : "border-green-500/50 text-green-400"}
                >
                  <MapPin className="h-3 w-3 mr-1" />
                  Zona de recogida
                </Button>
                <Button
                  variant={newZoneMode === "service" ? "default" : "outline"}
                  size="sm"
                  onClick={() => setNewZoneMode("service")}
                  className={newZoneMode === "service" ? "bg-yellow-500 text-black" : "border-yellow-500/50 text-yellow-400"}
                >
                  <Globe className="h-3 w-3 mr-1" />
                  Zona de destino
                </Button>
              </div>
            </div>

            <Separator />

            {/* Add new zone - Hierarchical selector */}
            <div className="space-y-3">
              <Label className="text-sm font-semibold">Añadir nueva zona:</Label>
              
              {/* Search */}
              <div className="relative">
                <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 h-4 w-4 text-muted-foreground" />
                <Input
                  placeholder="Buscar cantón, distrito o municipio..."
                  value={searchTerm}
                  onChange={(e) => setSearchTerm(e.target.value)}
                  className="pl-10"
                />
              </div>

              {/* Quick zone buttons */}
              <div className="space-y-2">
                <Label className="text-xs text-muted-foreground">Selección rápida:</Label>
                <div className="flex flex-wrap gap-2">
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={() => quickAddZone("Liechtenstein", "country", newZoneMode)}
                    className="text-xs border-orange-500/50 text-orange-400 hover:bg-orange-500/10"
                  >
                    <Globe className="h-3 w-3 mr-1" />
                    Todo Liechtenstein
                  </Button>
                  {SWISS_CANTONS.slice(0, 6).map((canton) => (
                    <Button
                      key={canton.id}
                      variant="outline"
                      size="sm"
                      onClick={() => quickAddZone(canton.name, "canton", newZoneMode)}
                      className="text-xs border-purple-500/50 text-purple-400 hover:bg-purple-500/10"
                    >
                      {canton.code}
                    </Button>
                  ))}
                </div>
              </div>

              {/* Hierarchical selector */}
              <div className="border rounded-lg p-4 space-y-3">
                <div className="grid grid-cols-1 md:grid-cols-3 gap-3">
                  {/* Canton selector */}
                  <div>
                    <Label className="text-xs text-muted-foreground flex items-center gap-1">
                      <Building className="h-3 w-3" />
                      Cantón
                    </Label>
                    <Select value={selectedCanton} onValueChange={(v) => {
                      setSelectedCanton(v);
                      setSelectedDistrict("");
                      setSelectedMunicipality("");
                      setNewExclusions([]);
                    }}>
                      <SelectTrigger className="mt-1">
                        <SelectValue placeholder="Seleccionar cantón" />
                      </SelectTrigger>
                      <SelectContent className="max-h-64">
                        <SelectItem value="liechtenstein">🇱🇮 Liechtenstein</SelectItem>
                        {filteredCantons.map((canton) => (
                          <SelectItem key={canton.id} value={canton.id}>
                            {canton.code} - {canton.name}
                          </SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                  </div>

                  {/* District selector */}
                  <div>
                    <Label className="text-xs text-muted-foreground flex items-center gap-1">
                      <MapPin className="h-3 w-3" />
                      Distrito / Región
                    </Label>
                    <Select 
                      value={selectedDistrict} 
                      onValueChange={(v) => {
                        setSelectedDistrict(v);
                        setSelectedMunicipality("");
                        setNewExclusions([]);
                      }}
                      disabled={!selectedCanton}
                    >
                      <SelectTrigger className="mt-1">
                        <SelectValue placeholder={selectedCanton ? "Todo el cantón" : "Selecciona cantón primero"} />
                      </SelectTrigger>
                      <SelectContent className="max-h-64">
                        <SelectItem value="__all__">Todo el cantón</SelectItem>
                        {availableDistricts.map((district) => (
                          <SelectItem key={district.id} value={district.id}>
                            {district.name}
                          </SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                  </div>

                  {/* Municipality selector */}
                  <div>
                    <Label className="text-xs text-muted-foreground flex items-center gap-1">
                      <MapPin className="h-3 w-3" />
                      Municipio
                    </Label>
                    <Select 
                      value={selectedMunicipality} 
                      onValueChange={setSelectedMunicipality}
                      disabled={!selectedDistrict || selectedDistrict === "__all__"}
                    >
                      <SelectTrigger className="mt-1">
                        <SelectValue placeholder={selectedDistrict && selectedDistrict !== "__all__" ? "Todo el distrito" : "Selecciona distrito"} />
                      </SelectTrigger>
                      <SelectContent className="max-h-64">
                        <SelectItem value="__all__">Todo el distrito</SelectItem>
                        {availableMunicipalities.map((municipality) => (
                          <SelectItem key={municipality} value={municipality}>
                            {municipality}
                          </SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                  </div>
                </div>

                {/* Custom zone name */}
                <div>
                  <Label className="text-xs text-muted-foreground">O escribe una zona personalizada:</Label>
                  <Input
                    placeholder="Ej: Región del Rin, Zona fronteriza..."
                    value={customZoneName}
                    onChange={(e) => setCustomZoneName(e.target.value)}
                    className="mt-1"
                  />
                </div>

                {/* Preview */}
                {(selectedCanton || customZoneName) && (
                  <div className="p-3 rounded-lg bg-muted/30 border border-dashed">
                    <Label className="text-xs text-muted-foreground">Zona a añadir:</Label>
                    <div className="flex items-center gap-2 mt-1">
                      <Badge variant="outline" className={newZoneMode === 'pickup' ? 'border-green-500 text-green-400' : 'border-yellow-500 text-yellow-400'}>
                        {newZoneMode === 'pickup' ? 'Recogida' : 'Destino'}
                      </Badge>
                      <Badge variant="outline">{getZoneType()}</Badge>
                      <span className="font-medium">{getZoneName()}</span>
                    </div>
                  </div>
                )}

                {/* Exclusions */}
                {(selectedCanton || customZoneName) && (
                  <Collapsible open={showExclusions} onOpenChange={setShowExclusions}>
                    <CollapsibleTrigger asChild>
                      <Button variant="outline" size="sm" className="w-full">
                        {showExclusions ? <ChevronDown className="h-4 w-4 mr-2" /> : <ChevronRight className="h-4 w-4 mr-2" />}
                        Excluir lugares específicos ({newExclusions.length} seleccionados)
                      </Button>
                    </CollapsibleTrigger>
                    <CollapsibleContent className="mt-3">
                      <div className="p-3 rounded-lg bg-red-500/5 border border-red-500/20">
                        <Label className="text-xs text-muted-foreground mb-2 block">
                          Selecciona los lugares que quieres EXCLUIR de esta zona:
                        </Label>
                        <div className="max-h-48 overflow-y-auto">
                          <div className="flex flex-wrap gap-2">
                            {(selectedCanton === "liechtenstein" ? LIECHTENSTEIN.municipalities : availableMunicipalities).map((place) => (
                              <Button
                                key={place}
                                variant={newExclusions.includes(place) ? "default" : "outline"}
                                size="sm"
                                onClick={() => {
                                  if (newExclusions.includes(place)) {
                                    setNewExclusions(newExclusions.filter(p => p !== place));
                                  } else {
                                    setNewExclusions([...newExclusions, place]);
                                  }
                                }}
                                className={newExclusions.includes(place) ? "bg-red-500 text-white text-xs" : "text-xs"}
                              >
                                {place}
                                {newExclusions.includes(place) && <X className="ml-1 h-3 w-3" />}
                              </Button>
                            ))}
                          </div>
                        </div>
                      </div>
                    </CollapsibleContent>
                  </Collapsible>
                )}

                <Button
                  onClick={handleAddZone}
                  disabled={saving || !getZoneName()}
                  className="w-full bg-yellow-400 text-black hover:bg-yellow-500"
                >
                  {saving ? <Loader2 className="h-4 w-4 animate-spin mr-2" /> : <Plus className="h-4 w-4 mr-2" />}
                  Añadir zona {newExclusions.length > 0 && `(${newExclusions.length} exclusiones)`}
                </Button>
              </div>
            </div>

            {/* All cantons list */}
            <details className="group">
              <summary className="cursor-pointer text-sm text-muted-foreground hover:text-foreground flex items-center gap-2">
                <ChevronDown className="h-4 w-4 group-open:rotate-180 transition-transform" />
                Ver todos los cantones y distritos
              </summary>
              <div className="mt-3 max-h-96 overflow-y-auto border rounded-lg p-3 space-y-2">
                {/* Liechtenstein */}
                <Collapsible
                  open={expandedCantons.has("liechtenstein")}
                  onOpenChange={() => toggleCantonExpansion("liechtenstein")}
                >
                  <CollapsibleTrigger className="flex items-center gap-2 w-full p-2 rounded hover:bg-muted/30 cursor-pointer">
                    {expandedCantons.has("liechtenstein") ? 
                      <ChevronDown className="h-4 w-4" /> : <ChevronRight className="h-4 w-4" />}
                    <Globe className="h-4 w-4 text-orange-400" />
                    <span className="font-medium">🇱🇮 Liechtenstein</span>
                    <Badge variant="outline" className="text-xs ml-auto">País</Badge>
                  </CollapsibleTrigger>
                  <CollapsibleContent>
                    <div className="ml-6 mt-2 flex flex-wrap gap-1">
                      <Button
                        variant="ghost"
                        size="sm"
                        onClick={() => quickAddZone("Liechtenstein", "country", newZoneMode)}
                        className="text-xs text-orange-400"
                      >
                        Todo Liechtenstein
                      </Button>
                      {LIECHTENSTEIN.municipalities.map((municipality) => (
                        <Button
                          key={municipality}
                          variant="ghost"
                          size="sm"
                          onClick={() => quickAddZone(municipality, "municipality", newZoneMode)}
                          className="text-xs"
                        >
                          {municipality}
                        </Button>
                      ))}
                    </div>
                  </CollapsibleContent>
                </Collapsible>

                {/* Swiss cantons */}
                {SWISS_CANTONS.map((canton) => (
                  <Collapsible
                    key={canton.id}
                    open={expandedCantons.has(canton.id)}
                    onOpenChange={() => toggleCantonExpansion(canton.id)}
                  >
                    <CollapsibleTrigger className="flex items-center gap-2 w-full p-2 rounded hover:bg-muted/30 cursor-pointer">
                      {expandedCantons.has(canton.id) ? 
                        <ChevronDown className="h-4 w-4" /> : <ChevronRight className="h-4 w-4" />}
                      <Building className="h-4 w-4 text-purple-400" />
                      <span className="font-medium">{canton.code}</span>
                      <span className="text-sm text-muted-foreground">{canton.name}</span>
                      <Badge variant="outline" className="text-xs ml-auto">{canton.districts.length} distritos</Badge>
                    </CollapsibleTrigger>
                    <CollapsibleContent>
                      <div className="ml-6 mt-2 space-y-1">
                        <Button
                          variant="ghost"
                          size="sm"
                          onClick={() => quickAddZone(canton.name, "canton", newZoneMode)}
                          className="text-xs text-purple-400 w-full justify-start"
                        >
                          Todo el cantón
                        </Button>
                        {canton.districts.map((district) => (
                          <div key={district.id} className="flex items-center gap-2">
                            <Button
                              variant="ghost"
                              size="sm"
                              onClick={() => quickAddZone(district.name, "district", newZoneMode)}
                              className="text-xs text-blue-400 flex-1 justify-start"
                            >
                              <MapPin className="h-3 w-3 mr-1" />
                              {district.name}
                            </Button>
                            <Badge variant="outline" className="text-xs">{district.municipalities.length}</Badge>
                          </div>
                        ))}
                      </div>
                    </CollapsibleContent>
                  </Collapsible>
                ))}
              </div>
            </details>
          </div>
        )}

        {/* ROUTES TAB */}
        {activeTab === "routes" && (
          <div className="space-y-4">
            {/* Info */}
            <div className="p-3 rounded-lg bg-blue-500/10 border border-blue-500/30 text-sm">
              <p className="text-blue-400 font-medium mb-1">💡 Rutas específicas</p>
              <p className="text-blue-300 text-xs">
                Define rutas punto a punto. Si estás en el origen, aparecerás para clientes que quieran ir al destino.
              </p>
            </div>

            {/* Current routes */}
            {routes.length > 0 && (
              <div className="space-y-2">
                <Label className="text-sm">Tus rutas ({routes.length}):</Label>
                <div className="max-h-48 overflow-y-auto space-y-2">
                  {routes.map((route) => (
                    <div
                      key={route.id}
                      className="flex items-center justify-between p-3 rounded-lg border border-border bg-muted/20"
                    >
                      <div className="flex items-center gap-2 text-sm">
                        <Badge variant="outline" className="text-xs">
                          {route.originType}
                        </Badge>
                        <span>{route.origin}</span>
                        <span className="text-yellow-400">→</span>
                        <span className="font-medium">{route.destination}</span>
                        <Badge variant="outline" className="text-xs">
                          {route.destType}
                        </Badge>
                      </div>
                      <Button
                        variant="ghost"
                        size="sm"
                        onClick={() => handleDeleteRoute(route.id)}
                        className="text-destructive"
                      >
                        <X className="h-4 w-4" />
                      </Button>
                    </div>
                  ))}
                </div>
              </div>
            )}

            <Separator />

            {/* Add new route */}
            <div className="space-y-3">
              <Label className="text-sm font-semibold">Añadir nueva ruta:</Label>
              
              <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
                <div>
                  <Label className="text-xs text-muted-foreground">Origen</Label>
                  <div className="flex gap-2 mt-1">
                    <Select value={newOriginType} onValueChange={setNewOriginType}>
                      <SelectTrigger className="w-24">
                        <SelectValue />
                      </SelectTrigger>
                      <SelectContent>
                        <SelectItem value="city">Ciudad</SelectItem>
                        <SelectItem value="region">Región</SelectItem>
                        <SelectItem value="canton">Cantón</SelectItem>
                        <SelectItem value="place">Lugar</SelectItem>
                      </SelectContent>
                    </Select>
                    <Input
                      placeholder={baseCity}
                      value={newOrigin}
                      onChange={(e) => setNewOrigin(e.target.value)}
                      className="flex-1"
                    />
                  </div>
                </div>
                <div>
                  <Label className="text-xs text-muted-foreground">Destino</Label>
                  <div className="flex gap-2 mt-1">
                    <Select value={newDestType} onValueChange={setNewDestType}>
                      <SelectTrigger className="w-24">
                        <SelectValue />
                      </SelectTrigger>
                      <SelectContent>
                        <SelectItem value="city">Ciudad</SelectItem>
                        <SelectItem value="region">Región</SelectItem>
                        <SelectItem value="canton">Cantón</SelectItem>
                        <SelectItem value="place">Lugar</SelectItem>
                      </SelectContent>
                    </Select>
                    <Input
                      placeholder="Ej: Aeropuerto Zúrich"
                      value={newDestination}
                      onChange={(e) => setNewDestination(e.target.value)}
                      className="flex-1"
                    />
                  </div>
                </div>
              </div>

              <Button
                onClick={handleAddRoute}
                disabled={saving || !newOrigin || !newDestination}
                className="w-full bg-yellow-400 text-black hover:bg-yellow-500"
              >
                {saving ? <Loader2 className="h-4 w-4 animate-spin mr-2" /> : <Plus className="h-4 w-4 mr-2" />}
                Añadir ruta
              </Button>
            </div>

            {/* Popular places for routes */}
            <details className="group">
              <summary className="cursor-pointer text-sm text-muted-foreground hover:text-foreground flex items-center gap-2">
                <ChevronDown className="h-4 w-4 group-open:rotate-180 transition-transform" />
                Lugares populares para rutas
              </summary>
              <div className="mt-3 space-y-3">
                {/* Airports */}
                <div>
                  <Label className="text-xs text-muted-foreground flex items-center gap-1">
                    <Plane className="h-3 w-3" />
                    Aeropuertos
                  </Label>
                  <div className="flex flex-wrap gap-2 mt-1">
                    {POPULAR_PLACES.airports.map((airport) => (
                      <Button
                        key={airport.id}
                        variant="outline"
                        size="sm"
                        onClick={() => setNewDestination(airport.name)}
                        className="text-xs"
                      >
                        {airport.code} - {airport.name}
                      </Button>
                    ))}
                  </div>
                </div>

                {/* Train stations */}
                <div>
                  <Label className="text-xs text-muted-foreground flex items-center gap-1">
                    <Train className="h-3 w-3" />
                    Estaciones de tren
                  </Label>
                  <div className="flex flex-wrap gap-2 mt-1">
                    {POPULAR_PLACES.trainStations.map((station) => (
                      <Button
                        key={station.id}
                        variant="outline"
                        size="sm"
                        onClick={() => setNewDestination(station.name)}
                        className="text-xs"
                      >
                        {station.name}
                      </Button>
                    ))}
                  </div>
                </div>

                {/* Ski resorts */}
                <div>
                  <Label className="text-xs text-muted-foreground flex items-center gap-1">
                    <Mountain className="h-3 w-3" />
                    Estaciones de esquí
                  </Label>
                  <div className="flex flex-wrap gap-2 mt-1">
                    {POPULAR_PLACES.skiResorts.map((resort) => (
                      <Button
                        key={resort.id}
                        variant="outline"
                        size="sm"
                        onClick={() => setNewDestination(resort.name)}
                        className="text-xs"
                      >
                        {resort.name}
                      </Button>
                    ))}
                  </div>
                </div>
              </div>
            </details>
          </div>
        )}
      </CardContent>

      {/* Edit Zone Modal */}
      <Dialog open={showEditModal} onOpenChange={setShowEditModal}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Editar exclusiones</DialogTitle>
            <DialogDescription>
              Selecciona los lugares que quieres excluir de "{editingZone?.zoneName}"
            </DialogDescription>
          </DialogHeader>
          <div className="space-y-4">
            <div className="flex flex-wrap gap-2 max-h-64 overflow-y-auto">
              {editingZone && (editingZone.zoneType === 'country' ? LIECHTENSTEIN.municipalities : getMunicipalitiesByCanton(
                SWISS_CANTONS.find(c => c.name === editingZone.zoneName)?.id || ''
              )).map((place: string) => (
                <Button
                  key={place}
                  variant={editExclusions.includes(place) ? "default" : "outline"}
                  size="sm"
                  onClick={() => {
                    if (editExclusions.includes(place)) {
                      setEditExclusions(editExclusions.filter(p => p !== place));
                    } else {
                      setEditExclusions([...editExclusions, place]);
                    }
                  }}
                  className={editExclusions.includes(place) ? "bg-red-500 text-white text-xs" : "text-xs"}
                >
                  {place}
                  {editExclusions.includes(place) && <X className="ml-1 h-3 w-3" />}
                </Button>
              ))}
            </div>
            <div className="flex gap-2 justify-end">
              <Button variant="outline" onClick={() => setShowEditModal(false)}>
                Cancelar
              </Button>
              <Button 
                onClick={handleUpdateZoneExclusions} 
                disabled={saving}
                className="bg-yellow-400 text-black hover:bg-yellow-500"
              >
                {saving ? "Guardando..." : "Guardar"}
              </Button>
            </div>
          </div>
        </DialogContent>
      </Dialog>
    </Card>
  );
}
